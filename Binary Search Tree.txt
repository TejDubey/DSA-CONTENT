15 10 20 5

// You are using GCC
#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* left;
    struct Node* right;
};

struct Node* newNode(int data) {
    struct Node* node = (struct Node*)malloc(sizeof(struct Node));
    node->data = data;
    node->left = node->right = NULL;
    return node;
}

struct Node* insert(struct Node* root, int data) {
    if (root == NULL) {
        return newNode(data);
    }

    if (data > root->data) {
        root->right = insert(root->right, data);
    } else if (data <= root->data) {
        root->left = insert(root->left, data);
    }

    return root;
}

void levelOrderTraversal(struct Node* root, int level, int* product, int* isFirst) {
    if (root == NULL) {
        return;
    }

    if (level == 1) {
        printf("%d ", root->data);

        if (*isFirst) {
            *product *= root->data;
            *isFirst = 0;
        }
    } else {
        levelOrderTraversal(root->left, level - 1, product, isFirst);
        levelOrderTraversal(root->right, level - 1, product, isFirst);
    }
}

int height(struct Node* node) {
    if (node == NULL) {
        return 0;
    }

    int leftHeight = height(node->left);
    int rightHeight = height(node->right);

    return 1+((leftHeight > rightHeight) ? leftHeight : rightHeight);
}

int main() {
    int N;
    scanf("%d", &N);
    int values[N+1];
    
    for (int i = 1; i <= N; i++) {
        scanf("%d", &values[i]);

    for (int i = 1; i <= N; i++) {
        root = insert(root, values[i]);
    }

    int treeHeight = height(root);

    int product = 1;

    for (int level = 1; level <= treeHeight; level++) {
        int isFirst = 1;
        levelOrderTraversal(root, level, &product, &isFirst);
    }

    printf("\n");
    printf("%d\n", product);

    return 0;
}

    int predecessor = -1;
    while (root != NULL) {
        if (root->data < k) {

            root = root->left;

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
int findmin(struct TreeNode*t){
    while(t->left!=NULL){
        t=t->left;
    }
    return t->val;
}
struct TreeNode* deleteNode(struct TreeNode* root, int key) {
    if(root==NULL) return NULL;
    else if(root->val==key){
        if(root->left==NULL && root->right==NULL){ //0 child 
            free(root);
            return NULL;
        }
        else if(root->left!=NULL && root->right==NULL){ //1 child left child
            struct TreeNode*t=root->left;
            free(root);
            return t;
        }
        else if(root->left==NULL && root->right!=NULL){ //1 child right child
            struct TreeNode*t=root->right;
            free(root);
            return t;
        }
        else if(root->left!=NULL && root->right!=NULL){ //2 child left and right
            int min=findmin(root->right);
            root->val=min;
            root->right=deleteNode(root->right,min);
        }
    }
    else if(root->val>key){
        root->left=deleteNode(root->left,key);
    }
    else if(root->val<key){
        root->right=deleteNode(root->right,key);
    }
    return root;
}
