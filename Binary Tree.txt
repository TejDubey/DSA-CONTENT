Binary Tree

Q1. You are given a task to check whether a given tree is a binary tree, you need to ensure that it satisfies the properties of a binary tree. 
Here are the key properties of a binary tree:
Each node in the tree has at most two children.
For each node, all nodes in its left subtree have values less than the node's value, and all nodes in its right subtree have values greater than the node's value.
Your task is to write a program to implement the same.

Sample test cases :
Input 1 :
5 3 7 -1 -1 -1 -1
Output 1 :
The given tree is a Binary Tree.
Input 2 :
10 5 15 -1 12 -1 -1 -1 -1
Output 2 :
The given tree is not a Binary Tree.


#include <stdio.h>
#include <limits.h>
#include <stdlib.h>
struct Node{
    int data;
    struct Node *left,*right;
};
struct Queue{
    struct Node* nodes[100];
    int front,rear;
};
int isempty(struct Queue* queue){
    return queue->front==queue->rear;
}
void initialise(struct Queue* queue){
    queue->front=queue->rear=0;
}
void enqueue(struct Queue*queue,struct Node*node){
    if(node!=NULL)
    queue->nodes[queue->rear++]=node;
}
struct Node*dequeue(struct Queue*queue){
    return queue->nodes[queue->front++];
}
struct Node* createNode(int data){
    struct Node*node=(struct Node*)malloc(sizeof(struct Node));
    node->data=data;
    node->left=NULL;
    node->right=NULL;
    return node;
}
struct Node* buildTree(int arr[], int n){
    if(n==0||arr[0]==-1){
        return NULL;
    }
    struct Queue queue;
    struct Node* root=createNode(arr[0]);
    initialise(&queue);
    enqueue(&queue,root);
    int i=1;
    while(!isempty(&queue) && i<n){
        struct Node*curr=dequeue(&queue);
        if (arr[i] != -1) curr->left = createNode(arr[i]);
        enqueue(&queue, curr->left);
        i++;
        if (i < n && arr[i] != -1) curr->right = createNode(arr[i]);
        enqueue(&queue, curr->right);
        i++;
    }
    return root;
}
int isBST(struct Node*root, int min, int max){
    if (root == NULL)
        return 1;
    if (root->data <= min || root->data >= max)
        return 0;
    return isBST(root->left, min, root->data) && isBST(root->right, root->data, max);
}
int main(){
    int arr[100];
    int n=0;
    while(scanf("%d ",&arr[n])!=-1){
        n++;
    }
    struct Node*root=buildTree(arr,n);
    if(isBST(root, INT_MIN, INT_MAX)){
        printf("The given tree is a Binary Tree.");
    }
    else{
        printf("The given tree is not a Binary Tree.");
    }
}

Q3. Problem Statement
Noah wants to create a binary tree and print its zig-zag level order traversal. He needs your help to implement this.
Write a program that takes input from Noah and builds a binary tree using level order insertion. Then, the program should print the zig-zag level order traversal of the binary tree.

Input 1 :
7
1 2 3 7 6 5 4
Output 1 :
1 3 2 7 6 5 4 
Input 2 :
4
1 6 2 3
Output 2 :
1 2 6 3 
Input 3 :
15
1 2 3 7 6 5 4 9 8 11 12 13 14 15 16
Output 3 :
1 3 2 7 6 5 4 16 15 14 13 12 11 8 9

#include <stdio.h>
#include <stdlib.h>
struct Node {
    int data;
    struct Node *left;
    struct Node *right;
};
struct Queue {
    struct Node* nodes[100];
    int front;
    int rear;
};
void initQueue(struct Queue* q) {
    q->front = 0;
    q->rear = 0;
}
int isEmpty(struct Queue* q) {
    return q->front == q->rear;
}
void enqueue(struct Queue* q, struct Node* node) {
    if (node != NULL) {
        q->nodes[q->rear++] = node;
    }
}
struct Node* dequeue(struct Queue* q) {
    return q->nodes[q->front++];
}
struct Node* createNode(int data) {
    struct Node* node = (struct Node*)malloc(sizeof(struct Node));
    node->data = data;
    node->left = NULL;
    node->right = NULL;
    return node;
}
struct Node* buildTree(int arr[], int n) {
    if (n == 0) {
        return NULL;
    }
    struct Node* root = createNode(arr[0]);
    struct Queue q;
    initQueue(&q);
    enqueue(&q, root);
    int i = 1;
    while (!isEmpty(&q) && i < n) {
        struct Node* curr = dequeue(&q);

        if (i < n) {
            curr->left = createNode(arr[i++]);
            enqueue(&q, curr->left);
        }

        if (i < n) {
            curr->right = createNode(arr[i++]);
            enqueue(&q, curr->right);
        }
    }
    return root;
}
void zigzagTraversal(struct Node* root) {
    if (root == NULL) {
        return;
    }
    struct Queue q;
    initQueue(&q);
    enqueue(&q, root);
    int level = 0;
    while (!isEmpty(&q)) {
        int size = q.rear - q.front;
        int temp[size];
        for (int i = 0; i < size; i++) {
            struct Node* node = dequeue(&q);
            temp[i] = node->data;
            if (node->left) {
                enqueue(&q, node->left);
            }
            if (node->right) {
                enqueue(&q, node->right);
            }
        }
        if (level % 2 == 0) {
            for (int i = 0; i < size; i++) {
                printf("%d ", temp[i]);
            }
        } else {
            for (int i = size - 1; i >= 0; i--) {
                printf("%d ", temp[i]);
            }
        }
        level++;
    }
}
int main() {
    int n;
    scanf("%d", &n);
    int arr[100];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    struct Node* root = buildTree(arr, n);
    zigzagTraversal(root);
    return 0;
}

Q4. Arjun is developing a program to analyze binary trees built from arrays. He wants to:

Construct a binary tree from an array using level order insertion (array index i has left child at 2*i+1 and right child at 2*i+2).
Perform an in-order traversal of the tree and store the result in an array.
For each node in the original input order, print its in-order predecessor and successor based on the in-order traversal.
If the predecessor or successor does not exist, print 0 in its place.


Help Arjun to implement the task.

Input format :
The first line of input consists of an integer n, representing the number of nodes in the binary tree.

The second line of input consists of n space-separated integers, representing the node values of the binary tree in level-order.

Output format :
For each element in the original input array:

Print one line in the format:

<node_value> : <predecessor_value>,<successor_value>

If the predecessor or successor does not exist, print 0 in its place.



Refer to the sample output for the formatting specifications.

Code constraints :
In this scenario, the given test cases will fall under the following constraints:

1 ≤ n ≤ 20

1 ≤ node values ≤ 1000

Sample test cases :
Input 1 :
4 
1 2 3 4
Output 1 :
1 : 2,3
2 : 4,1
3 : 1,0
4 : 0,2
Input 2 :
7
10 5 12 4 7 11 13
Output 2 :
10 : 7,11
5 : 4,7
12 : 11,13
4 : 0,5
7 : 5,10
11 : 10,12
13 : 12,0
Note :
The program will be evaluated only after the “Submit Code” is clicked.
Extra spaces and new line characters in the program output will result in the failure of the test case.

#include <stdio.h>
#include <stdlib.h>
struct Node {
    int data;
    struct Node *left, *right;
};
struct Node* buildTree(int arr[], int n, int i) {
    if (i >= n) return NULL;
    struct Node* node = (struct Node*)malloc(sizeof(struct Node));
    node->data = arr[i];
    node->left = buildTree(arr, n, 2*i + 1);
    node->right = buildTree(arr, n, 2*i + 2);
    return node;
}
void inorderTraversal(struct Node* root, int inorder[], int *index) {
    if (!root) return;
    inorderTraversal(root->left, inorder, index);
    inorder[(*index)++] = root->data;
    inorderTraversal(root->right, inorder, index);
}
int main() {
    int n;
    scanf("%d", &n);
    int arr[20];
    for (int i = 0; i < n; i++) scanf("%d", &arr[i]);
    struct Node* root = buildTree(arr, n, 0);
    int inorder[20], inIndex = 0;
    inorderTraversal(root, inorder, &inIndex);
    for (int i = 0; i < n; i++) {
        int val = arr[i], pred = 0, succ = 0;
        for (int j = 0; j < n; j++) {
            if (inorder[j] == val) {
                if (j > 0) pred = inorder[j-1];
                if (j < n-1) succ = inorder[j+1];
                break;
            }
        }
        printf("%d : %d,%d\n", val, pred, succ);
    }
    return 0;
}
