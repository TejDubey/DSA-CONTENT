Q1. In a college programming competition, participants are tasked with developing a program to reverse the elements in a circular linked list containing N nodes. 

Input 1 :
7
98 15 26 36 49 51 62
Output 1 :
62 51 49 36 26 15 98 
Input 2 :
3
14 29 35
Output 2 :
35 29 14 

#include <stdio.h>
#include <stdlib.h>
struct Node{
    int data;
    struct Node*next;
};
void addnode(struct Node** head, int val){
    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));
    newnode->data=val;
    if(*head==NULL){
        *head=newnode;
        (*head)->next=*head;
        return;
    }
    struct Node* temp=*head;
    while(temp->next!=*head){
        temp=temp->next;
    }
    temp->next=newnode;
    newnode->next=*head;
}
void insertnode(struct Node** head, int val, int pos, int n){
    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));
    newnode->data=val;
    struct Node* temp=*head;
    for(int i=1;i<pos-1 && temp!=NULL;i++){
        temp=temp->next;
    }
    newnode->next=temp->next;
    temp->next=newnode;
}
void reverse(struct Node** head){
    struct Node* prev=NULL;
    struct Node* curr=*head;
    struct Node* next=NULL;
    struct Node* first=*head;
    do{
        next=curr->next;
        curr->next=prev;
        prev=curr;
        curr=next;
    }
    while(curr!=*head);
    first->next=prev; 
    //first =*head = first -> 1
    //1->next=prev(prev=3 after do while loop)
    // head->1->2->3->NULL reverse 3->2->1->
    //3->2->1->3
    *head=prev; //now head update karna hai
    //prev=3 *head=prev (*head node becomes prev i.e 3)
}
void display(struct Node* head){
    if(head==NULL) return;
    struct Node*temp=head;
    do{
        printf("%d ",temp->data);
        temp=temp->next;
    }
    while(temp!=head);
}
int main(){
    int n;
    scanf("%d",&n);
    struct Node* head=NULL;
    for(int i=0;i<n;i++){
        int y;
        scanf("%d",&y);
        addnode(&head,y);
    }
    reverse(&head);
    display(head);
}

Q3. Write a program to implement a Circular Linked List that stores the names of currently running applications. The program should accept application names as input and allow iteration over them. If no applications are provided, it should indicate that there are no running applications.

The first line of output prints "Circular Linked List - Running Applications"

If applications were added:

The next line prints "Running Applications:"
Followed by each application name on a new line in the order they were added.
If no applications were added before typing exit:

It directly prints "No applications to iterate."

Input 1 :
Microsoft Office Suite
Adobe Creative Cloud
Google Chrome
Mozilla Firefox
VLC Media Player
Skype
exit
Output 1 :
Circular Linked List - Running Applications

Running Applications:
Microsoft Office Suite
Adobe Creative Cloud
Google Chrome
Mozilla Firefox
VLC Media Player
Skype
Input 2 :
exit
Output 2 :
Circular Linked List - Running Applications

No applications to iterate.


#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Node {
    char data[101];
    struct Node* next;
};

void addnode(struct Node** head, char val[]) {
    struct Node* newnode = (struct Node*)malloc(sizeof(struct Node));
    strcpy(newnode->data, val);

    if (*head == NULL) {
        *head = newnode;
        newnode->next = *head;
        return;
    }

    struct Node* temp = *head;
    while (temp->next != *head) {
        temp = temp->next;
    }
    temp->next = newnode;
    newnode->next = *head;
}

void display(struct Node* head) {
    if (head == NULL) {
        printf("No applications to iterate.\n");
        return;
    }

    printf("Running Applications:\n");
    struct Node* temp = head;
    do {
        printf("%s\n", temp->data);
        temp = temp->next;
    } while (temp != head);
}

int main() {
    struct Node* head = NULL;
    char app_name[101];

    while (1) {
        fgets(app_name, sizeof(app_name), stdin);

        size_t len = strlen(app_name);
        if (len > 0 && app_name[len - 1] == '\n') {
            app_name[len - 1] = '\0';
        }

        if (strcmp(app_name, "exit") == 0) {
            break;
        }

        addnode(&head, app_name);
    }

    printf("Circular Linked List - Running Applications\n\n");
    display(head);

    return 0;
}

Q5. Imagine a circular conveyor belt where each container has an ID number, and the sum of its digits reflects specific traits. Your task is to remove all containers with an even-digit sum, leaving only those with an odd-digit sum to proceed. The conveyor belt is modelled as a circular linked list, where each node represents a container. 

You need to efficiently filter out these nodes while preserving the circular structure.

Input 1 :
6
9 11 34 6 13 21
Output 1 :
9 34 21 
Input 2 :
5
34 56 78 92 36
Output 2 :
34 56 78 92 36 


int digitsum(int x){
    int sum=0;
    while(x!=0){
        int temp=x%10;
        sum+=temp;
        x/=10;
    }
    return sum;
}
void removeEvenDigitSumNodes(struct Node** head){
    struct Node*temp=*head;
    struct Node*prev=NULL;
    if(*head==NULL) return;
    while(digitsum((*head)->data)%2==0){
        if(temp->next==*head){
            *head=NULL; 
            free(temp);
            return;
        }
        struct Node*last=*head;
        while(last->next!=*head){
            last=last->next;
        }
        temp=*head;
        last->next=(*head)->next;
        *head=(*head)->next;
        free(temp);
    }
    prev = *head;
    temp = (*head)->next;
    while (temp != *head) {
        if (digitsum(temp->data) % 2 == 0) {
            prev->next = temp->next;
            free(temp);
            temp = prev->next;
        } else {
            prev = temp;
            temp = temp->next;
        }
    }
}
void display(struct Node* head){
    struct Node* temp=head;
    do{
        printf("%d ",temp->data);
        temp=temp->next;
    }while(temp!=head);
}

Q6. In a coding competition, participants are challenged to implement a function to split a circular linked list into two separate circular lists, after inserting all the given elements at the beginning of the list. 

Students must develop a program that receives a circular linked list and outputs two separate circular lists.

If there is an even number of elements, split it into two halves with the same count.
If there is an odd number of elements, split it in such a way that the first half can have more elements than the other. 


Input 1 :
8
5 6 3 2 8 9 4 1
Output 1 :
1 4 9 8 
2 3 6 5 
Input 2 :
5
1 2 3 4 5
Output 2 :
5 4 3 
2 1 



#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
};

struct Node* createNode(int val) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    newNode->next = newNode;
    return newNode;
}

void insertBeginning(struct Node** head, int val) {
    struct Node* newNode = createNode(val);
    if (*head == NULL) {
        *head = newNode;
        return;
    }
    struct Node* temp = *head;
    while (temp->next != *head)
        temp = temp->next;
    newNode->next = *head;
    temp->next = newNode;
    *head = newNode;
}

void printList(struct Node* head) {
    if (head == NULL) return;
    struct Node* temp = head;
    do {
        printf("%d ", temp->data);
        temp = temp->next;
    } while (temp != head);
    printf("\n");
}

void splitList(struct Node* head, struct Node** firstHalf, struct Node** secondHalf, int n) {
    if (!head) return;
    int firstCount = (n + 1) / 2;
    int secondCount = n - firstCount;
    struct Node* temp = head;
    int i;

    for (i = 0; i < firstCount; i++) {
        struct Node* newNode = createNode(temp->data);
        if (*firstHalf == NULL)
            *firstHalf = newNode;
        else {
            struct Node* t = *firstHalf;
            while (t->next != *firstHalf)
                t = t->next;
            t->next = newNode;
            newNode->next = *firstHalf;
        }
        temp = temp->next;
    }

    for (i = 0; i < secondCount; i++) {
        struct Node* newNode = createNode(temp->data);
        if (*secondHalf == NULL)
            *secondHalf = newNode;
        else {
            struct Node* t = *secondHalf;
            while (t->next != *secondHalf)
                t = t->next;
            t->next = newNode;
            newNode->next = *secondHalf;
        }
        temp = temp->next;
    }
}

int main() {
    int n, i, val;
    struct Node* head = NULL;
    struct Node* firstHalf = NULL;
    struct Node* secondHalf = NULL;

    scanf("%d", &n);
    for (i = 0; i < n; i++) {
        scanf("%d", &val);
        insertBeginning(&head, val);
    }

    splitList(head, &firstHalf, &secondHalf, n);

    printList(firstHalf);
    printList(secondHalf);

    return 0;
}
