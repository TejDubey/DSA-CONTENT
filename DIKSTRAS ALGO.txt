// You are using GCC
#include <stdio.h>
#include <stdlib.h>

#define INF 1000000000

typedef struct Node {
    int v, w;
    struct Node* next;
} Node;

Node* adj[105];
int dist[105];
int visited[105];

Node* createNode(int v, int w) {
    Node* node = (Node*)malloc(sizeof(Node));
    node->v = v;
    node->w = w;
    node->next = NULL;
    return node;
}

int main() {
    int V, E;
    scanf("%d", &V);
    scanf("%d", &E);

    for (int i = 0; i < V; i++) {
        adj[i] = NULL;
        dist[i] = INF;
        visited[i] = 0;
    }

    for (int i = 0; i < E; i++) {
        int u, v, w;
        scanf("%d %d %d", &u, &v, &w);

        Node* n1 = createNode(v, w);
        n1->next = adj[u];
        adj[u] = n1;

        Node* n2 = createNode(u, w);
        n2->next = adj[v];
        adj[v] = n2;
    }

    int S;
    scanf("%d", &S);

    dist[S] = 0;

    for (int i = 0; i < V; i++) {
        int u = -1, dmin = INF;

        for (int j = 0; j < V; j++) {
            if (!visited[j] && dist[j] < dmin) {
                dmin = dist[j];
                u = j;
            }
        }

        if (u == -1) break;
        visited[u] = 1;

        Node* temp = adj[u];
        while (temp != NULL) {
            int v = temp->v;
            int w = temp->w;
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
            }
            temp = temp->next;
        }
    }

    printf("Shortest distances from vertex %d to all other vertices: ", S);
    for (int i = 0; i < V; i++) {
        printf("%d ", dist[i]);
    }

    return 0;
}

#include <stdio.h>
#include <limits.h>

#define MAX 105
#define INF 1000000000

int adj[MAX][MAX];
int dist[MAX];
int visited[MAX];
int parent[MAX];

int main() {
    int n, m;
    scanf("%d", &n);
    scanf("%d", &m);

    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            adj[i][j] = (i == j ? 0 : INF);

    for (int i = 0; i < n; i++) {
        dist[i] = INF;
        visited[i] = 0;
        parent[i] = -1;
    }

    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d %d %d", &u, &v, &w);
        adj[u][v] = w;
        adj[v][u] = w;
    }

    int src, dest;
    scanf("%d", &src);
    scanf("%d", &dest);

    dist[src] = 0;

    for (int count = 0; count < n; count++) {
        int u = -1, best = INF;
        for (int i = 0; i < n; i++) {
            if (!visited[i] && dist[i] < best) {
                best = dist[i];
                u = i;
            }
        }

        if (u == -1) break;
        visited[u] = 1;

        for (int v = 0; v < n; v++) {
            if (!visited[v] && adj[u][v] < INF) {
                if (dist[u] + adj[u][v] < dist[v]) {
                    dist[v] = dist[u] + adj[u][v];
                    parent[v] = u;
                }
            }
        }
    }

    if (dist[dest] == INF) {
        printf("No path found");
        return 0;
    }

    int path[MAX], len = 0;
    int cur = dest;
    while (cur != -1) {
        path[len++] = cur;
        cur = parent[cur];
    }

    printf("Shortest path: ");
    for (int i = len - 1; i >= 0; i--) {
        printf("%d", path[i]);
        if (i > 0) printf(" -> ");
    }
    printf("\n");

    printf("Shortest distance: %d", dist[dest]);

    return 0;
}

// You are using GCC
#include <stdio.h>
#include <limits.h>

#define INF 1000000000

typedef struct {
    int v, w;
} Edge;

Edge adj[1005][10005];
int edgeCount[1005];

int dist[1005];
int parent[1005];
int visited[1005];

int main() {
    int N, M;
    scanf("%d %d", &N, &M);

    for (int i = 0; i < N; i++) {
        edgeCount[i] = 0;
        dist[i] = INF;
        visited[i] = 0;
        parent[i] = -1;
    }

    for (int i = 0; i < M; i++) {
        int u, v, w;
        scanf("%d %d %d", &u, &v, &w);
        adj[u][edgeCount[u]].v = v;
        adj[u][edgeCount[u]].w = w;
        edgeCount[u]++;
    }

    int S;
    scanf("%d", &S);

    dist[S] = 0;

    for (int k = 0; k < N; k++) {
        int u = -1, best = INF;

        for (int i = 0; i < N; i++) {
            if (!visited[i] && dist[i] < best) {
                best = dist[i];
                u = i;
            }
        }

        if (u == -1) break;
        visited[u] = 1;

        for (int i = 0; i < edgeCount[u]; i++) {
            int v = adj[u][i].v;
            int w = adj[u][i].w;

            if (!visited[v] && dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                parent[v] = u;
            }
        }
    }

    for (int v = 0; v < N; v++) {
        if (v == S) continue;

        if (dist[v] == INF) continue;

        printf("%d ", dist[v]);

        int path[1005], len = 0, cur = v;
        while (cur != -1) {
            path[len++] = cur;
            cur = parent[cur];
        }

        for (int i = len - 1; i >= 0; i--) {
            printf("%d", path[i]);
            if (i > 0) printf(" -> ");
        }

        printf("\n");
    }

    return 0;
}
