#include <stdio.h>

int islandSize;
int dx[4] = {1, -1, 0, 0};

void dfs(int x, int y) {
    visited[x][y] = 1;
    islandSize++;

    for (int i = 0; i < 4; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];

        if (nx >= 0 && ny >= 0 && nx < n && ny < n &&
            grid[nx][ny] == 1 && !visited[nx][ny]) {
            dfs(nx, ny);
        }
    }
}

int getIslandSize(int x, int y) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            visited[i][j] = 0;

    islandSize = 0;
    dfs(x, y);
    return islandSize;
}

int main() {
    scanf("%d", &n);

    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%d", &grid[i][j]);

    int maxIsland = 0;
    int allLand = 1;

    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            if (grid[i][j] == 0)
                allLand = 0;

    if (allLand) {
        printf("%d", n * n);
        return 0;
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {

            if (grid[i][j] == 0) {
                grid[i][j] = 1;

                for (int a = 0; a < n; a++) {
                    for (int b = 0; b < n; b++) {
                        if (grid[a][b] == 1) {
                            int size = getIslandSize(a, b);

                    }
                }

            }
        }

    return 0;

int n,m;

int dx[8]={1,-1,0,0,1,1,-1,-1};
int dy[8]={0,0,1,-1,1,-1,1,-1};

int dfs(int x,int y,char c){
    int best=1;
    for(int k=0;k<8;k++){
        int nx=x+dx[k], ny=y+dy[k];
        if(nx>=0 && ny>=0 && nx<n && ny<m){
            if(a[nx][ny]==c+1){
