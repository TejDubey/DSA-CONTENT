Q1. Linked List:
Rotate Linked List left by k positions

input:
3
6 4
3 6 9 12 15 18
6 2
2 4 6 8 10 12
6 3
4 8 12 16 20 24

output:
15 18 3 6 9 12 
6 8 10 12 2 4 
16 20 24 4 8 12

#include <stdio.h>
#include <stdlib.h>
struct Node{
    int data;
    struct Node* next;
};
void addnode(struct Node** head,int val){
    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));
    newnode->data=val;
    newnode->next=NULL;
    if(*head==NULL){
        *head=newnode;
        return;
    }
    struct Node* temp=*head;
    while(temp->next!=NULL){
        temp=temp->next;
    }
    temp->next=newnode;
}
struct Node* rotateleft(struct Node* head, int k){
    if(k==0 || head==NULL){
        return head;
    }
    struct Node* curr=head;
    for(int i=1;i<k;i++){
        curr=curr->next;
    }
    if(curr->next==NULL) return head;
    struct Node* newhead=curr->next;
    curr->next=NULL;
    struct Node* tail=newhead;
    while(tail->next!=NULL){
        tail=tail->next;
    }
    tail->next=head;
    return newhead;
}
void display(struct Node* head){
    struct Node* temp=head;
    while(temp!=NULL){
        printf("%d ",temp->data);
        temp=temp->next;
    }
    printf("\n");
}
int main(){
    int x;
    scanf("%d",&x);
    for(int i=0;i<x;i++){
        struct Node*head=NULL;
        int n,k;
        scanf("%d %d",&n,&k);
        for(int i=0;i<n;i++){
            int y;
            scanf("%d",&y);
            addnode(&head,y);
        }
        struct Node* newhead=rotateleft(head,k);
        display(newhead);
    }
}

Q2. Sort LL in aesc order 

input:
5
78 89 34 51 67

output:
34 51 67 78 89

#include <stdio.h>
#include <stdlib.h>
struct Node{
    int data;
    struct Node* next;
};
void addnode(struct Node** head,int val){
    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));
    newnode->data=val;
    newnode->next=NULL;
    if(*head==NULL){
        *head=newnode;
        return;
    }
    struct Node* temp=*head;
    while(temp->next!=NULL){
        temp=temp->next;
    }
    temp->next=newnode;
}
void sort(struct Node* head){
    if(head==NULL) return;
    struct Node* i;
    struct Node* j;
    for(i=head;i->next!=NULL;i=i->next){
        for(j=i->next;j!=NULL;j=j->next){
            if(i->data > j->data){
                int temp=i->data;
                i->data=j->data;
                j->data=temp;
            }
        }
    }
}
void display(struct Node* head){
    struct Node* temp=head;
    while(temp!=NULL){
        printf("%d ",temp->data);
        temp=temp->next;
    }
}
int main(){
    struct Node*head=NULL;
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        int y;
        scanf("%d",&y);
        addnode(&head,y);
    }
    sort(head);
    display(head);
}

Q3. insert even numbers at the front in reverse order and then insert odd numbers at end in reverse order

input:
5
12 15 13 14 16

output:
16 14 12 13 15

#include <stdio.h>
#include <stdlib.h>
struct Node{
    int data;
    struct Node*next;
};
void addnode(struct Node** head, int val){
    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));
    newnode->data=val;
    newnode->next=NULL;
    if(*head==NULL){
        *head=newnode;
        return;
    }
    struct Node* temp=*head;
    while(temp->next!=NULL){
        temp=temp->next;
    }
    temp->next=newnode;
}
void reverse(struct Node** head){
    struct Node* prev=NULL;
    struct Node* curr=*head;
    struct Node* next=NULL;
    while(curr!=0){
        next=curr->next;
        curr->next=prev;
        prev=curr;
        curr=next;
    }
    *head=prev;
}
void print(struct Node*head1, struct Node* head2){
    struct Node* temp1=head1;
    struct Node* temp2=head2;
    while(temp1!=NULL){
        printf("%d ",temp1->data);
        temp1=temp1->next;
    }
    while(temp2!=NULL){
        printf("%d ",temp2->data);
        temp2=temp2->next;
    }
}
int main(){
    struct Node*head1=NULL;
    struct Node*head2=NULL;
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        int y;
        scanf("%d",&y);
        if(y%2==0){
            addnode(&head1,y);
        }
        else{
            addnode(&head2,y);
        }
    }
    reverse(&head1);
    reverse(&head2);
    print(head1,head2);
}

#include <stdio.h>
#include <stdlib.h>
struct Node{
    int data;
    struct Node*next;
};
void insertbegin(struct Node** head,int x){
    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));
    newnode->data=x;
    newnode->next=*head;
    *head=newnode;
}
void insertend(struct Node** head,int x){
    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));
    newnode->data=x;
    newnode->next=NULL;
    if(*head==NULL){
        *head=newnode;
        return;
    }
    struct Node*temp=*head;
    while(temp->next!=NULL){
        temp=temp->next;
    }  
    temp->next=newnode;
}
void insertatpos(struct Node** head,int pos,int x){
    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));
    newnode->data=x;
    if(pos==1){
        newnode->next=NULL;
        *head=newnode;
        return;
    }
    struct Node*temp=*head;
    for(int i=1;temp!=NULL && i<pos-1;i++){
        temp=temp->next;
    }
    newnode->next=temp->next;
    temp->next=newnode;
}
void display(struct Node*head){
    struct Node*temp=head;
    while(temp!=NULL){
        printf("%d ",temp->data);
        temp=temp->next;
    }
    printf("\n");
}
