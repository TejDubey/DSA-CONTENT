            printf("%d",temp->coeff);

Q2. Timothy wants to evaluate polynomial expressions for his mathematics homework. He needs a program that allows him to input the coefficients of a polynomial based on its degree and compute the polynomial's value for a given input of x. Implement a function that takes the degree, coefficients, and the value of x, and returns the evaluated result of the polynomial.

Example

Input:
degree of the polynomial = 2
coefficient of x2 = 13
coefficient of x1 = 12
coefficient of x0 = 11
x = 1
Output:
36


#include <stdio.h>
#include <stdlib.h>
#include <math.h>
struct Node {
    int coeff;
    int pwr;
    struct Node* next;
};

void addnode(struct Node** head, int coeff, int pwr) {
    struct Node* newnode = (struct Node*)malloc(sizeof(struct Node));
    newnode->coeff = coeff;
    newnode->pwr = pwr;
    newnode->next = NULL;
    if (*head == NULL) {
        *head = newnode;
        return;
    }
    struct Node* temp = *head;
    while (temp->next != NULL)
        temp = temp->next;
    temp->next = newnode;
}

int evaluate(struct Node* head, int x) {
    int sum = 0;
    struct Node* temp = head;
    while(temp!=0){
        sum=sum+(temp->coeff*((int)pow(x,temp->pwr)));
        temp=temp->next;
    }
    return sum;
}

int main() {
    int degree;
    scanf("%d", &degree);
    struct Node* poly = NULL;
    for (int i = degree; i >= 0; i--) {
        int coeff;
        scanf("%d", &coeff);
        addnode(&poly, coeff, i);
    }
    int x;
    scanf("%d", &x);
    int result = evaluate(poly, x);
    printf("%d", result);
    return 0;
}

    struct Node* newnode=(struct Node*)malloc(sizeof(struct Node));

void addition(struct Node*pol1,struct Node*pol2,struct Node**pol3){
    while(pol1!=NULL && pol2!=NULL){
        if(pol1->pwr>pol2->pwr){
            addnode(pol3,pol1->coeff,pol1->pwr);
            pol1=pol1->next;
        }
        else if(pol2->pwr>pol1->pwr){
            addnode(pol3,pol2->coeff,pol2->pwr);
            pol2=pol2->next;
        }
        else{
            addnode(pol3,pol1->coeff+pol2->coeff,pol1->pwr);
            pol2=pol2->next;
            pol1=pol1->next;
        }
    }
    while (pol1 != NULL) {
        addnode(pol3, pol1->coeff, pol1->pwr);
        pol1 = pol1->next;
    }
    while (pol2 != NULL) {
        addnode(pol3, pol2->coeff, pol2->pwr);
        pol2 = pol2->next;
    }
}

#include <stdio.h>

struct Term {
    int coeff;
    int exp;
};

int main() {
    int n, m, delete_exp;
    scanf("%d", &n);
    struct Term poly1[n];
    for(int i = 0; i < n; i++)
        scanf("%d %d", &poly1[i].coeff, &poly1[i].exp);

    scanf("%d", &m);
    struct Term poly2[m];
    for(int i = 0; i < m; i++)
        scanf("%d %d", &poly2[i].coeff, &poly2[i].exp);

    scanf("%d", &delete_exp);

    struct Term result[50];
    int res_count = 0;

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            int coeff = poly1[i].coeff * poly2[j].coeff;
            int exp = poly1[i].exp + poly2[j].exp;
            int found = 0;
            for(int k = 0; k < res_count; k++) {
                if(result[k].exp == exp) {
                    result[k].coeff += coeff;
                    found = 1;
                    break;
                }
            }
            if(!found) {
                result[res_count].coeff = coeff;
                result[res_count].exp = exp;
                res_count++;
            }
        }
    }

    printf("Result of the multiplication: ");
    int first = 1;
    for(int i = 0; i < res_count; i++) {
        if(result[i].coeff == 0) continue;
        if(!first) printf(" + ");
        if(result[i].exp == 0) printf("%d", result[i].coeff);
        else if(result[i].exp == 1) printf("%dx", result[i].coeff);
        else printf("%dx^%d", result[i].coeff, result[i].exp);
        first = 0;
    }
    printf("\n");

    printf("Result after deleting the term: ");
    first = 1;
    for(int i = 0; i < res_count; i++) {
        if(result[i].exp == delete_exp || result[i].coeff == 0) continue;
        if(!first) printf(" + ");
        if(result[i].exp == 0) printf("%d", result[i].coeff);
        else if(result[i].exp == 1) printf("%dx", result[i].coeff);
        else printf("%dx^%d", result[i].coeff, result[i].exp);
        first = 0;
    }
    printf("\n");

    return 0;
}

#include <stdio.h>
#include <stdlib.h>

struct Node {
    long long coeff;
    int pwr;
    struct Node* next;
};

void insertTerm(struct Node** head, long long coeff, int pwr) {
    if (coeff == 0) return;
    struct Node* newnode = (struct Node*)malloc(sizeof(struct Node));
    newnode->coeff = coeff;
    newnode->pwr = pwr;
    newnode->next = NULL;
    if (*head == NULL || (*head)->pwr < pwr) {
        newnode->next = *head;
        *head = newnode;
        return;
    }
    struct Node* temp = *head;
    struct Node* prev = NULL;
    while (temp != NULL && temp->pwr > pwr) {
        prev = temp;
        temp = temp->next;
    }
    if (temp != NULL && temp->pwr == pwr) {
        temp->coeff += coeff;
        free(newnode);
    } else {
        newnode->next = temp;
        if (prev != NULL) prev->next = newnode;
    }
}

struct Node* multiply(struct Node* poly1, struct Node* poly2) {
    struct Node* result = NULL;
    for (struct Node* p1 = poly1; p1 != NULL; p1 = p1->next) {
        for (struct Node* p2 = poly2; p2 != NULL; p2 = p2->next) {
            insertTerm(&result, p1->coeff * p2->coeff, p1->pwr + p2->pwr);
        }
    }
    return result;
}

void display(struct Node* head) {
    struct Node* temp = head;
    int first = 1;
    while (temp != NULL) {
        if (!first) {
            if (temp->coeff >= 0) printf(" + ");
            else printf(" - ");
        } else {
            if (temp->coeff < 0) printf("-");
        }
        long long coeff = temp->coeff >= 0 ? temp->coeff : -temp->coeff;
        if (temp->pwr > 1) printf("%lldx^%d", coeff, temp->pwr);
        else if (temp->pwr == 1) printf("%lldx", coeff);
        else printf("%lld", coeff);
        first = 0;
        temp = temp->next;
    }
    printf("\n");
}

int main() {
    int n, m;
    long long coeff;
    int pwr;
    struct Node* poly1 = NULL;
    struct Node* poly2 = NULL;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%lld %d", &coeff, &pwr);
        insertTerm(&poly1, coeff, pwr);
    }
    scanf("%d", &m);
    for (int i = 0; i < m; i++) {
        scanf("%lld %d", &coeff, &pwr);
        insertTerm(&poly2, coeff, pwr);
    }
    display(poly1);
    display(poly2);
    struct Node* result = multiply(poly1, poly2);
    display(result);
    return 0;
}
