// You are using GCC
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int u, v;
    int w;
} Edge;

typedef struct {
    int parent;
    int rank;
} DSU;

int cmpEdges(const void *a, const void *b) {
    const Edge *e1 = (const Edge *)a;
    const Edge *e2 = (const Edge *)b;
    if (e1->w != e2->w) return (e1->w < e2->w) ? -1 : 1;
    if (e1->u != e2->u) return (e1->u < e2->u) ? -1 : 1;
    return (e1->v < e2->v) ? -1 : (e1->v > e2->v);
}

int findSet(DSU dsu[], int x) {
    if (dsu[x].parent != x)
        dsu[x].parent = findSet(dsu, dsu[x].parent);
    return dsu[x].parent;
}

void unionSet(DSU dsu[], int a, int b) {
    a = findSet(dsu, a);
    b = findSet(dsu, b);
    if (a == b) return;
    if (dsu[a].rank < dsu[b].rank) {
        dsu[a].parent = b;
    } else if (dsu[a].rank > dsu[b].rank) {
        dsu[b].parent = a;
    } else {
        dsu[b].parent = a;
        dsu[a].rank++;
    }
}

int main() {
    int V, E;
    if (scanf("%d", &V) != 1) return 0;
    if (scanf("%d", &E) != 1) return 0;

    Edge *edges = (Edge *)malloc(sizeof(Edge) * E);
    for (int i = 0; i < E; i++) {
        scanf("%d %d %d", &edges[i].u, &edges[i].v, &edges[i].w);
    }

    qsort(edges, E, sizeof(Edge), cmpEdges);

    DSU *dsu = (DSU *)malloc(sizeof(DSU) * V);
    for (int i = 0; i < V; i++) {
        dsu[i].parent = i;
        dsu[i].rank = 0;
    }

    Edge *mst = (Edge *)malloc(sizeof(Edge) * (V - 1));
    long long minCost = 0;
    int cnt = 0;

    for (int i = 0; i < E && cnt < V - 1; i++) {
        int a = findSet(dsu, edges[i].u);
        int b = findSet(dsu, edges[i].v);
        if (a != b) {
            unionSet(dsu, a, b);
            mst[cnt++] = edges[i];
            minCost += edges[i].w;
        }
    }

    printf("Following are the edges in the constructed MST\n");
    for (int i = 0; i < cnt; i++) {
        printf("%d -- %d == %d\n", mst[i].u, mst[i].v, mst[i].w);
    }
    printf("Minimum Cost Spanning Tree: %lld", minCost);

    free(edges);
    free(dsu);
    free(mst);
    return 0;
}

#include <stdio.h>

typedef struct {
    int u, v;
    int w;
} Edge;

int cmp(const void *a, const void *b) {
    return ((Edge*)a)->w - ((Edge*)b)->w;
}

int find(int parent[], int x) {
    if (parent[x] != x)
        parent[x] = find(parent, parent[x]);
    return parent[x];
}

void unite(int parent[], int rank[], int a, int b) {
    a = find(parent, a);
    b = find(parent, b);
    if (a == b) return;

    if (rank[a] < rank[b]) parent[a] = b;
    else if (rank[a] > rank[b]) parent[b] = a;
    else {
        parent[b] = a;
        rank[a]++;
    }
}

int main() {
    int N;
    scanf("%d", &N);

    int cost[25][25];
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            scanf("%d", &cost[i][j]);

    Edge edges[400];
    int eCount = 0;

    for (int i = 0; i < N; i++) {
        for (int j = i + 1; j < N; j++) {
            if (cost[i][j] != 0 && cost[i][j] != 9999) {
                edges[eCount].u = i;
                edges[eCount].v = j;
                edges[eCount].w = cost[i][j];
                eCount++;
            }
        }
    }

    qsort(edges, eCount, sizeof(Edge), cmp);

    int parent[25], rank[25] = {0};
    for (int i = 0; i < N; i++) parent[i] = i;

    int chosen = 0;
    for (int i = 0; i < eCount && chosen < N - 1; i++) {
        int u = edges[i].u;
        int v = edges[i].v;

        if (find(parent, u) != find(parent, v)) {
            unite(parent, rank, u, v);
            printf("%d -> %d\n", u + 1, v + 1);
            chosen++;
        }
    }

    return 0;
}

// You are using GCC
#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

typedef struct Edge {
    int to;
    int cost;
    struct Edge* next;
} Edge;

typedef struct {
    int node;
    int cost;
} HeapNode;

typedef struct {
    HeapNode* arr;
    int size;
} MinHeap;

void push(MinHeap* h, int node, int cost) {
    h->arr[++h->size].node = node;
    h->arr[h->size].cost = cost;
    int i = h->size;

    while (i > 1 && h->arr[i].cost < h->arr[i/2].cost) {
        HeapNode temp = h->arr[i];
        h->arr[i] = h->arr[i/2];
        h->arr[i/2] = temp;
        i /= 2;
    }
}

HeapNode pop(MinHeap* h) {
    HeapNode top = h->arr[1];
    h->arr[1] = h->arr[h->size--];

    int i = 1;
    while (1) {
        int left = i * 2;
        int right = left + 1;
        int smallest = i;

        if (left <= h->size && h->arr[left].cost < h->arr[smallest].cost)
            smallest = left;
        if (right <= h->size && h->arr[right].cost < h->arr[smallest].cost)
            smallest = right;

        if (smallest != i) {
            HeapNode temp = h->arr[i];
            h->arr[i] = h->arr[smallest];
            h->arr[smallest] = temp;
            i = smallest;
        } else break;
    }

    return top;
}

int main() {
    int numCities, numConnections;
    scanf("%d", &numCities);
    scanf("%d", &numConnections);

    Edge** graph = (Edge**)calloc(numCities, sizeof(Edge*));

    for (int i = 0; i < numConnections; i++) {
        int u, v, w;
        scanf("%d %d %d", &u, &v, &w);

        Edge* e1 = (Edge*)malloc(sizeof(Edge));
        e1->to = v; e1->cost = w; e1->next = graph[u];
        graph[u] = e1;

        Edge* e2 = (Edge*)malloc(sizeof(Edge));
        e2->to = u; e2->cost = w; e2->next = graph[v];
        graph[v] = e2;
    }

    int* visited = (int*)calloc(numCities, sizeof(int));
    MinHeap heap;
    heap.arr = (HeapNode*)malloc((numConnections + 5) * sizeof(HeapNode));
    heap.size = 0;

    int totalCost = 0;
    push(&heap, 0, 0);

    while (heap.size > 0) {
        HeapNode top = pop(&heap);
        int node = top.node;
        int cost = top.cost;

        if (visited[node]) continue;
        visited[node] = 1;
        totalCost += cost;

        Edge* temp = graph[node];
        while (temp) {
            if (!visited[temp->to]) {
                push(&heap, temp->to, temp->cost);
            }
            temp = temp->next;
        }
    }

    printf("%d", totalCost);
    return 0;
}

int minKey(int key[], int mstSet[], int n) {
    int min = INT_MAX, index = -1;
    for (int i = 0; i < n; i++) {
        if (mstSet[i] == 0 && key[i] < min) {
            min = key[i];
            index = i;
        }
    }
    return index;
}

int main() {
    int n;
    scanf("%d", &n);

    int cost[n][n];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &cost[i][j]);
        }
    }

    int parent[n];
    int key[n];
    int mstSet[n];

    for (int i = 0; i < n; i++) {
        key[i] = INT_MAX;
        mstSet[i] = 0;
    }

    key[0] = 0;
    parent[0] = -1;

    for (int count = 0; count < n - 1; count++) {
        int u = minKey(key, mstSet, n);
        mstSet[u] = 1;

        for (int v = 0; v < n; v++) {
            if (cost[u][v] != 0 && mstSet[v] == 0 && cost[u][v] < key[v]) {
                parent[v] = u;
                key[v] = cost[u][v];
            }
        }
    }

    int mstMatrix[n][n];
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            mstMatrix[i][j] = 0;

    int totalCost = 0;

    for (int i = 1; i < n; i++) {
        int u = parent[i];
        int v = i;
        mstMatrix[u][v] = cost[u][v];
        mstMatrix[v][u] = cost[u][v];
        totalCost += cost[u][v];
    }

    printf("Spanning Tree Matrix:\n");
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            printf("%d ", mstMatrix[i][j]);
        }
        printf("\n");
    }

    printf("Total Cost: %d", totalCost);

    return 0;
}
